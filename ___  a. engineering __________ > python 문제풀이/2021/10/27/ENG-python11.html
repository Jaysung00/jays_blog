# 내 풀이 -> OK

n = 5

def solution(n):
	answer = ''
	while n > 0:
		if n % 3 != 0:
			answer = str(n % 3) + answer
			n = n // 3
		else:
			answer = '4' + answer
			n = n // 3 - 1
	return answer

print(solution(n)) #Return 11

def change124(n):
	num = ['1','2','4']
	answer = ""


	while n > 0:
		n -= 1
		answer = num[n % 3] + answer
		n //= 3

	return answer
